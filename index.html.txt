        // --- EXPORT CSV ---
        function exportWeeklyCSV() {
            const now = new Date();
            const day = now.getDay() || 7;
            const monday = new Date(now);
            monday.setDate(now.getDate() - day + 1);
            monday.setHours(0,0,0,0);
            const startISO = monday.toISOString().split('T')[0];
            const weeklySales = salesHistory.filter(s => s.rawDate >= startISO);
            if(weeklySales.length === 0) return alert("Aucune vente enregistrée cette semaine.");
            let csvContent = "Date;Vendeur;Articles;Total\n";
            weeklySales.forEach(s => {
                const row = `${s.date};${s.seller};${s.items.replace(/,/g, '-')};${s.total}$`;
                csvContent += row + "\n";
            });
            const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
            const link = document.createElement("a");
            const url = URL.createObjectURL(blob);
            link.setAttribute("href", url);
            link.setAttribute("download", `compta_le_repere_semaine_${startISO}.csv`);
            link.style.visibility = 'hidden';
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }

        // --- CAISSE ---
        function addToCart(id) { const p = products.find(x=>x.id===id); if(p.stock <= 0) return alert("Rupture !"); cart.push({...p}); render(); }
        function clearCart() { cart = []; render(); }
        
        function validateSale() { 
            if(!cart.length) return; 
            const seller = prompt("Nom du vendeur :");
            if(!seller || seller.trim() === "") return;
            const sub = cart.reduce((a, b) => a + b.price, 0); 
            const tx = document.getElementById('tax-toggle').checked ? Math.round(sub*(taxConfig.rate/100)) : 0; 
            const disc = document.getElementById('discount-toggle').checked ? Math.round(sub*(discountConfig.rate/100)) : 0; 
            const tot = sub + tx - disc; 
            const entry = { date: new Date().toLocaleString('fr-FR'), rawDate: new Date().toISOString().split('T')[0], seller: seller.toUpperCase(), items: cart.map(i => i.name).join(', '), total: tot };
            salesHistory.unshift(entry);
            cart.forEach(i => products.find(x=>x.id===i.id).stock--); 
            money.weekly += tot; money.total += tot; cart = []; save(); render(); 
        }

        function updateTaxConfig() { taxConfig.name = document.getElementById('tax-name-input').value || "Taxe"; taxConfig.rate = parseInt(document.getElementById('tax-rate-input').value) || 0; save(); render(); }
        function updateDiscountConfig() { discountConfig.rate = parseInt(document.getElementById('discount-rate-input').value) || 0; save(); render(); }

            const sub = cart.reduce((a,b)=>a+b.price,0);
            const isT = document.getElementById('tax-toggle').checked;
            const isD = document.getElementById('discount-toggle').checked;
            const tA = isT && cart.length ? Math.round(sub*(taxConfig.rate/100)) : 0;
            const dA = isD && cart.length ? Math.round(sub*(discountConfig.rate/100)) : 0;

            if(tA) h += `<div class="text-[10px] text-blue-400 border-t border-gray-800 mt-2 pt-1 flex justify-between font-bold"><span>+ ${taxConfig.name}</span><span>${tA}$</span></div>`;
            if(dA) h += `<div class="text-[10px] text-red-500 flex justify-between font-bold italic"><span>- REMISE</span><span>${dA}$</span></div>`;
            document.getElementById('cart-items').innerHTML = h;
            document.getElementById('cart-total').innerText = (sub+tA-dA)+"$";

            const sF = document.getElementById('filter-start').value;
            const eF = document.getElementById('filter-end').value;
            let fS = salesHistory;
            if(sF) fS = fS.filter(s => s.rawDate >= sF);
            if(eF) fS = fS.filter(s => s.rawDate <= eF);
            document.getElementById('period-total').innerText = fS.reduce((a,b)=>a+b.total,0)+"$";
            document.getElementById('history-table').innerHTML = fS.map(s => `<tr class="border-b border-gray-800 hover:bg-white/5 transition"><td class="p-3 text-gray-500 font-mono">${s.date}</td><td class="text-blue-400 font-bold">${s.seller}</td><td class="lowercase text-gray-400">${s.items}</td><td class="text-right p-3 font-bold text-green-500">${s.total}$</td></tr>`).join('');

            document.getElementById('ca-weekly').innerText = money.weekly + "$";
            document.getElementById('ca-total').innerText = money.total + "$";
            document.getElementById('inventory-table').innerHTML = products.map(p => `<tr class="border-b border-gray-800 p-2"><td><img src="${p.img}" class="w-8 h-8 rounded object-cover"></td><td class="font-bold">${p.name}</td><td>${p.price}$</td><td class="text-center">${p.stock}</td><td class="text-right"><button onclick="editProductPrompt(${p.id})" class="text-blue-400 mr-2">Edit</button></td></tr>`).join('');

            document.getElementById('club-list').innerHTML = club.map(p => `<div class="bg-gray-900 p-2 rounded text-center"><img src="${p.img}" onclick="zoomImg('${p.img}')" class="w-full h-20 object-cover rounded mb-1 zoomable"><div class="text-[9px] font-bold text-blue-400">${p.name}</div><div class="text-[8px] text-gray-500">${p.role}</div></div>`).join('');
            document.getElementById('bar-menu-grid').innerHTML = products.map(p => `<div class="bg-dark-panel rounded overflow-hidden border border-gray-800"><img src="${p.img}" onclick="zoomImg('${p.img}')" class="w-full h-32 object-cover grayscale hover:grayscale-0 transition zoomable"><div class="p-2 flex justify-between"><h3>${p.name}</h3><span class="text-blue-400">${p.price}$</span></div></div>`).join('');

        }

        function editPerson(t) { const n = prompt("Nom :"); if(!n) return; const r = prompt("Rôle :"); const im = prompt("URL Photo :"); (t==='staff'?staff:club).push({id:Date.now(), name:n, role:r, img:im||"https://i.pravatar.cc/150"}); save(); render(); }
        function deletePerson(t,id) { if(confirm("Supprimer ?")) { if(t==='staff') staff=staff.filter(x=>x.id!==id); else club=club.filter(x=>x.id!==id); save(); render(); }}
        function editProductPrompt(id=null) { const p = id ? products.find(x=>x.id===id) : {name:"",price:0,img:""}; const n = prompt("Nom :", p.name); if(!n) return; const pr = parseInt(prompt("Prix :", p.price))||0; const im = prompt("URL :", p.img); if(id) Object.assign(products.find(x=>x.id===id), {name:n, price:pr, img:im}); else products.push({id:Date.now(), name:n, price:pr, stock:0, img:im}); save(); render(); }
